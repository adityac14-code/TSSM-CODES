def add_data(club_members):
  roll = int(input("Enter roll number: "))
  name = input("Enter name: ")

  # Find insertion position
  for i in range(len(club_members)):
    if roll < club_members[i][0]:
      # Insert at the found position
      club_members.insert(i, (roll, name))
      return club_members

  # Append to the end if no position found
  club_members.append((roll, name))
  return club_members

def display(club_members):
  for member in club_members:
    print(member)

def ternary_search(club_members):
  roll_no = int(input("Enter roll number to search: "))

  # Set search boundaries
  l = 0
  r = len(club_members) - 1

  while l <= r:
    # Calculate mid1 and mid2
    mid1 = l + (r - l) // 3
    mid2 = r - (r - l) // 3

    # Check for roll number at mid1 or mid2
    if roll_no == club_members[mid1][0]:
      print(f"Roll number {roll_no} belongs to {club_members[mid1][1]}")
      return
    elif roll_no == club_members[mid2][0]:
      print(f"Roll number {roll_no} belongs to {club_members[mid2][1]}")
      return

    # Update search range based on comparison
    if roll_no < club_members[mid1][0]:
      r = mid1 - 1
    elif roll_no > club_members[mid2][0]:
      l = mid2 + 1
    else:
      l = mid1 + 1
      r = mid2 - 1

  # Roll number not found
  print(f"Roll number {roll_no} not found. Add the name:")
  add_data(club_members)
  display(club_members)

def add_Elements(arr, n):
  """
  Adds elements to the array.

  Args:
    arr: The array to store the elements.
    n: The number of elements to add.

  Returns:
    The updated array with the added elements.
  """

  try:
    print("Enter Array Elements:")
    for _ in range(n):
      num = float(input())
      arr.append(num)
  except ValueError:
    print("Invalid input! Please enter valid numbers.")

  print(f"Array Elements: {arr}")
  return arr


def bubble_sort(arr, n):
  """
  Sorts the array using bubble sort.

  Args:
    arr: The array to sort.
    n: The number of elements in the array.

  Returns:
    None
  """

  for i in range(n):
    swapped = False
    for j in range(n - 1):
      if arr[j] > arr[j + 1]:
        arr[j], arr[j + 1] = arr[j + 1], arr[j]
        swapped = True

    if not swapped:
      break

  print("Sorted elements using bubble sort:")
  print(arr)

  print("Top 5 scores:", arr[-5:])


def selection_sort(arr, n):
  """
  Sorts the array using selection sort.

  Args:
    arr: The array to sort.
    n: The number of elements in the array.

  Returns:
    None
  """

  for i in range(n - 1):
    min_index = i
    for j in range(i + 1, n):
      if arr[j] < arr[min_index]:
        min_index = j

    arr[i], arr[min_index] = arr[min_index], arr[i]

  print("Sorted elements using selection sort:")
  print(arr)

  print("Top 5 scores:", arr[-5:])
